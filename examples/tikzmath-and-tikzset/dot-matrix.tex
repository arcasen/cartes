\documentclass{standalone}
\usepackage{tikz}

\usetikzlibrary{math}

\begin{document}

% begin_snippet
\begin{tikzpicture}
  \tikzmath{
    for \i in {0,1,2,3} {
      for \j in {0,1,2} {
        % 计算 x 和 y 坐标
        \x = \i * 1.25;  
        \y = \j * 1.25;  
        % 注意：绘图命令需要用花括号包围
        { \fill[red] (\x, \y) circle (1.5pt); }; 
      };
    };
  }
\end{tikzpicture}
% end_snippet

\end{document
