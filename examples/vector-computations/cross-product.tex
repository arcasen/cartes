\begin{tikzpicture}[
  x={({0.5*cos(-135)},{0.5*sin(-135)})}, % default: cm
  y={(1cm,0cm)}, 
  z={(0cm,1cm)}
]
  \tikzmath{
    \A{1} = 3; \A{2} = 0; \A{3} = 0;
    \B{1} = 0; \B{2} = 3; \B{3} = 0;
    \C{1} = 0; \C{2} = 0; \C{3} = 3;
  }

  \tikzset{
    space/cross={\A,\B,\C,\temp},
    space/normalize/origin={\temp,\n},
  }

  % 绘制坐标轴
  \draw[help lines,teal,-latex] (0,0,0) -- (1,0,0) node[right]{$x$};
  \draw[help lines,teal,-latex] (0,0,0) -- (0,1,0) node[below left]{$y$};
  \draw[help lines,teal,-latex] (0,0,0) -- (0,0,1) node[above]{$z$};

  \coordinate (A) at (\A{1},\A{2},\A{3});
  \coordinate (B) at (\B{1},\B{2},\B{3});
  \coordinate (C) at (\C{1},\C{2},\C{3});
  \coordinate (D) at (\n{1},\n{2},\n{3});

  \draw [thick,-latex] (A) -- (B);
  \draw [thick,-latex] (A) -- (C);
  \draw [thick,-latex,purple] (0,0,0) -- (D);
  
  \foreach \p/\placement in {A/below,B/below,C/above} {
    \fill[red] (\p) circle (1.5pt);
    \draw (\p) node[\placement] {$\p$};
  }
