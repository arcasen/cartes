\begin{tikzpicture}
  \tikzmath{
    \coefs{1} = 1; \coefs{2} = 1; \coefs{3} = 1; \coefs{4} = -2; \coefs{5} = -4; \coefs{6} = 0;
  }
  \tikzset{
    conics/define/quadratic={\coefs,\A},
    conics/rotate={\A,\B,\R,\ang},
    conics/translate={\B,\C,\T,\shift}
  }

  \typeout{========================}
  \typeout{Original Coefficient Matrix:}
  \typeout{\A{1,1},\A{1,2},\A{1,3}}
  \typeout{\A{2,1},\A{2,2},\A{2,3}}
  \typeout{\A{3,1},\A{3,2},\A{3,3}}
  \typeout{------------------------}
  \typeout{Rotation Matrix:}
  \typeout{\R{1,1},\R{1,2},\R{1,3}}
  \typeout{\R{2,1},\R{2,2},\R{2,3}}
  \typeout{\R{3,1},\R{3,2},\R{3,3}}
  \typeout{angle:\ang}
  \typeout{------------------------}
  \typeout{Coefficient Matrix after Rotation:}
  \typeout{\B{1,1},\B{1,2},\B{1,3}}
  \typeout{\B{2,1},\B{2,2},\B{2,3}}
  \typeout{\B{3,1},\B{3,2},\B{3,3}}
  \typeout{------------------------}
  \typeout{Translation Matrix:}
  \typeout{\T{1,1},\T{1,2},\T{1,3}}
  \typeout{\T{2,1},\T{2,2},\T{2,3}}
  \typeout{\T{3,1},\T{3,2},\T{3,3}}
  \typeout{xshift:\shift{1},yshift:\shift{2}}
  \typeout{------------------------}
  \typeout{Coefficient Matrix after Translation:}
  \typeout{\C{1,1},\C{1,2},\C{1,3}}
  \typeout{\C{2,1},\C{2,2},\C{2,3}}
  \typeout{\C{3,1},\C{3,2},\C{3,3}}
  \typeout{========================}

  \axes[grid] (-5:5,-5:5);
  \draw[rotate=\ang,shift={(\shift{1},\shift{2})}, thick,teal] (0,0) ellipse ({sqrt(-\C{3,3}/\C{1,1})} and {sqrt(-\C{3,3}/\C{2,2})});
\end{tikzpicture}
