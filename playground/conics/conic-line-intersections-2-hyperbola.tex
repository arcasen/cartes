\begin{tikzpicture}
  \tikzmath{
    \coefs{1} = 1; \coefs{2} = 1; \coefs{3} = -1; \coefs{4} = -2; \coefs{5} = -4; \coefs{6} = 0;
    % 两点的齐次坐标
    \U{1,1} = -1; \U{2,1} = -1; \U{3,1} = 1; 
    \V{1,1} = 4; \V{2,1} = 1; \V{3,1} = 1; 
  }
  \tikzset{
    conics/define/quadratic={\coefs,\A},
    conics/intersect cl={\A,\U,\V,\P,\Q},
    conics/dehomogenize={\U,\UU},
    conics/dehomogenize={\V,\VV},
    conics/dehomogenize={\P,\PP},
    conics/dehomogenize={\Q,\QQ},
  }

  \axes[grid] (-5:5,-5:5);
  \conic[draw=teal,thick] (\A);

  \coordinate (U) at (\UU{1},\UU{2});
  \coordinate (V) at (\VV{1},\VV{2});
  \coordinate (P) at (\PP{1},\PP{2});
  \coordinate (Q) at (\QQ{1},\QQ{2});

  \foreach \p/\placement in {P/above, Q/above, U/above, V/above}{
    \fill[red] (\p) circle (2pt);
    \draw (\p) node[\placement] {$\p$};
  }
   
\end{tikzpicture}
