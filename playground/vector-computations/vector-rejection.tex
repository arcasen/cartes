  \coordinate (A) at (\A{1},\A{2});
  \coordinate (B) at (\B{1},\B{2});
  \coordinate (C) at (\C{1},\C{2});
  \coordinate (D) at (\D{1},\D{2});

  \draw [thick,-latex] (A) -- (B);
  \draw [thick,-latex] (A) -- (C);
  \draw [thick,-latex,purple] (A) -- (D);

  \foreach \p/\placement in {A/below,B/below,C/above,D/below} {
    \fill[red] (\p) circle (2pt);
    \draw (\p) node[\placement] {$\p$};
  }
\end{tikzpicture}
