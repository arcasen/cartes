\begin{tikzpicture}
  \tikzmath{
    \A{1} = 4; \A{2} = 0;
    \B{1} = -1; \B{2} = -1;
    \C{1} = 0; \C{2} = 4;
    \A{1} = 4 * cos(120); \A{2} = 4 * sin(120);
    \B{1} = 4 * cos(210); \B{2} = 4 * sin(210);
    \C{1} = 4 * cos(-30); \C{2} = 4 * sin(-30);
  }

  \tikzset{
    plane/incenter={\A,\B,\C,\I},
    plane/project={\B,\I,\C,\D},
    plane/length={\I,\D,\r},
  }

  % 绘制坐标轴
  \draw[help lines,teal,-latex] (0,0) -- (1,0) node[right]{$x$};
  \draw[help lines,teal,-latex] (0,0) -- (0,1) node[below left]{$y$};

  \coordinate (A) at (\A{1},\A{2});
  \coordinate (B) at (\B{1},\B{2});
  \coordinate (C) at (\C{1},\C{2});
  \coordinate (I) at (\I{1},\I{2});

  \draw[thick] (A) -- (B) -- (C) -- cycle;
  \draw[teal] (I) circle (\r);

  \foreach \p/\placement in {A/below,B/below,C/above,I/above} {
    \fill[red] (\p) circle (2pt);
    \draw (\p) node[\placement] {$\p$};
  }
\end{tikzpicture}
